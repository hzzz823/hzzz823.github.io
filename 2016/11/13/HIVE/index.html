hive install

1 install hive

  1. brew install hive
  2. brew install mysql
  3. add /usr/local/Cellar/hive/2.1.0/libexec/lib in ~/.zshrc
  4. mysql -uroot
  5. CERATE USER 'hiveuser'@'localhost' IDENTIFIED BY 'password'
  6. GRANT ALL PRIVILEGES ON *.* TO 'hiveuser'@'localhost' IDENTIFIED BY 'password'
  7. download msyqlConnector curl -L 'http://www.mysql.com/get/Downloads/Connector-J/mysql-connector-java-5.1.40.tar.gz/from/http://mysql.he.net/' | tar xz
  8. cp mysql-connector-java-5.1.40/mysql-connector-java-5.1.40-bin.jar /usr/local/Cellar/hive/2.1.0/libexec/lib
  9. cd $HIVE_HOME/conf
  10. cp hive-default.xml.template hive-site.xml
  11. vi hive-site.xml
  12. hive
